// ====== DATA ====== let saldo = localStorage.getItem("saldo") || 100000000; let tugasSelesai = JSON.parse(localStorage.getItem("tugas")) || {}; let vip = localStorage.getItem("vip") || false; let riwayat = JSON.parse(localStorage.getItem("riwayat")) || []; // ====== UPDATE SALDO ====== function updateSaldo(){ document.querySelector(".balance").innerText = "Rp " + Number(saldo).toLocaleString(); } updateSaldo(); // ====== TUGAS ====== function tugas(n, el){ if(tugasSelesai[n]){ alert("Tugas sudah dikerjakan!"); return; } let reward = [0,5000,10000,15000,50000,100000]; // cek VIP if((n==4 || n==5) && vip!="true"){ alert("Harus VIP dulu!"); return; } saldo = Number(saldo) + reward[n]; tugasSelesai[n] = true; localStorage.setItem("saldo", saldo); localStorage.setItem("tugas", JSON.stringify(tugasSelesai)); // ubah tombol el.innerText = "✅ Selesai"; el.disabled = true; tambahRiwayat("Tugas "+n, reward[n]); updateSaldo(); } // ====== LOAD STATUS TUGAS ====== window.onload = function(){ updateSaldo(); for(let i=1;i<=5;i++){ if(tugasSelesai[i]){ let btn = document.querySelector(`[onclick="tugas(${i}, this)"]`); if(btn){ btn.innerText="✅ Selesai"; btn.disabled=true; } } } } // ====== DEPOSIT ====== function kirimDeposit(){ let jumlah = Number(document.getElementById("jumlahDepo").value); if(jumlah < 10000){ alert("Minimal deposit 10.000"); return; } saldo += jumlah; // VIP unlock if(jumlah >= 50000){ vip = true; localStorage.setItem("vip", true); alert("VIP aktif!"); } localStorage.setItem("saldo", saldo); tambahRiwayat("Deposit", jumlah); updateSaldo(); tutupDeposit(); } // ====== TARIK ====== function kirimTarik(){ let jumlah = Number(document.getElementById("jumlahTarik").value); if(jumlah > saldo){ alert("Saldo tidak cukup!"); return; } saldo -= jumlah; localStorage.setItem("saldo", saldo); tambahRiwayat("Tarik", -jumlah); updateSaldo(); tutupTarik(); } // ====== RIWAYAT ====== function tambahRiwayat(jenis, jumlah){ riwayat.push({ jenis: jenis, jumlah: jumlah, waktu: new Date().toLocaleString() }); localStorage.setItem("riwayat", JSON.stringify(riwayat)); tampilRiwayat(); } function tampilRiwayat(){ let box = document.getElementById("riwayatBox"); if(!box) return; box.innerHTML = ""; riwayat.slice().reverse().forEach(r => { box.innerHTML += `
${r.jenis}
Rp ${r.jumlah.toLocaleString()}
${r.waktu}
`; }); }
